@section Scripts {
    <script>
        let roomsTable;
        let currentView = 'table';

        $(document).ready(function() {
            initializeDataTable();
            initializeFilters();
            initializeViewToggle();
            reInitializeSelect2();
        });

        function reInitializeSelect2() {
            $('#createLocation').select2('destroy');
            $('#editLocation').select2('destroy');
            $('#createStatus').select2('destroy');
            $('#editStatus').select2('destroy');
            $('#newStatus').select2('destroy');

            $('#createLocation').select2({
                theme: 'bootstrap-5',
                placeholder: 'Select an option',
                dropdownParent: $('#createRoomModal'),
                allowClear: true
            });

            $('#editLocation').select2({
                theme: 'bootstrap-5',
                placeholder: 'Select an option',
                dropdownParent: $('#editRoomModal'),
                allowClear: true
            });

            $('#createStatus').select2({
                theme: 'bootstrap-5',
                placeholder: 'Select an option',
                dropdownParent: $('#createRoomModal'),
                allowClear: true
            });

            $('#editStatus').select2({
                theme: 'bootstrap-5',
                placeholder: 'Select an option',
                dropdownParent: $('#editRoomModal'),
                allowClear: true
            });

            $('#newStatus').select2({
                theme: 'bootstrap-5',
                placeholder: 'Select an option',
                dropdownParent: $('#changeStatusModal'),
                allowClear: true
            });
        }

        function initializeDataTable() {
            roomsTable = $('#roomsTable').DataTable({
                responsive: false,
                scrollX: true,
                pageLength: 25,
                order: [[6, 'desc']], // Sort by created date
                language: {
                    search: "",
                    searchPlaceholder: "üîç Search rooms...",
                    lengthMenu: "Show _MENU_ rooms",
                    info: "Showing _START_ to _END_ of _TOTAL_ rooms",
                    emptyTable: `
                        <div class="text-center py-4 text-muted">
                            <i class="fas fa-door-open fa-2x mb-2"></i>
                            <p class="mb-0">No rooms found</p>
                        </div>
                    `,
                    zeroRecords: `
                        <div class="text-center py-4 text-muted">
                            <i class="fas fa-search fa-2x mb-2"></i>
                            <p class="mb-0">No matching rooms found</p>
                        </div>
                    `,
                    paginate: {
                        previous: "‚Äπ",
                        next: "‚Ä∫"
                    }
                },
                dom:
                    "<'row mb-3'<'col-sm-12 col-md-6'l><'col-sm-12 col-md-6'f>>" +
                    "<'row'<'col-sm-12'tr>>" +
                    "<'row mt-3'<'col-sm-12 col-md-5'i><'col-sm-12 col-md-7'p>>",
            });
        }

        function initializeFilters() {
            $('#filterLocation, #filterStatus, #filterCapacity').on('change', applyFilters);
        }

        function initializeViewToggle() {
            $('.view-btn').on('click', function() {
                const view = $(this).data('view');
                switchView(view);
            });
        }

        function switchView(view) {
            currentView = view;
            $('.view-btn').removeClass('active');
            $(`.view-btn[data-view="${view}"]`).addClass('active');

            if (view === 'table') {
                $('#tableView').show();
                $('#cardView').hide();
            } else {
                $('#tableView').hide();
                $('#cardView').show();
            }
            applyFilters();
        }

        function applyFilters() {
            const location = $('#filterLocation').val();
            const status = $('#filterStatus').val();
            const capacity = $('#filterCapacity').val();

            if (currentView === 'table') {
                roomsTable.rows().every(function() {
                    const row = $(this.node());
                    let show = true;

                    if (location && row.data('location') != location) show = false;
                    if (status && row.data('status') !== status) show = false;
                    if (capacity) {
                        const cap = parseInt(row.data('capacity'));
                        if (capacity === '1-10' && (cap < 1 || cap > 10)) show = false;
                        if (capacity === '11-20' && (cap < 11 || cap > 20)) show = false;
                        if (capacity === '21-50' && (cap < 21 || cap > 50)) show = false;
                        if (capacity === '51+' && cap < 51) show = false;
                    }

                    if (show) {
                        row.show();
                    } else {
                        row.hide();
                    }
                });
            } else {
                $('.room-card-item').each(function() {
                    const card = $(this);
                    let show = true;

                    if (location && card.data('location') != location) show = false;
                    if (status && card.data('status') !== status) show = false;
                    if (capacity) {
                        const cap = parseInt(card.data('capacity'));
                        if (capacity === '1-10' && (cap < 1 || cap > 10)) show = false;
                        if (capacity === '11-20' && (cap < 11 || cap > 20)) show = false;
                        if (capacity === '21-50' && (cap < 21 || cap > 50)) show = false;
                        if (capacity === '51+' && cap < 51) show = false;
                    }

                    card.toggle(show);
                });
            }
        }

        function resetFilters() {
            $('#filterLocation, #filterStatus, #filterCapacity').val('').trigger('change');
            $('#filterLocation, #filterStatus, #filterCapacity').empty();
            if (currentView === 'table') {
                roomsTable.rows().every(function() {
                    $(this.node()).show();
                });
            } else {
                $('.room-card-item').show();
            }
        }

        function createRoom() {
            const data = {
                code: $('#createCode').val().trim(),
                name: $('#createName').val().trim(),
                locationId: $('#createLocation').val(),
                capacity: $('#createCapacity').val(),
                facilities: $('#createFacilities').val().trim(),
                status: $('#createStatus').val()
            };
            
            if (!data.code || !data.name || !data.locationId || !data.capacity || !data.status) {
                Swal.fire({
                    icon: 'error',
                    title: 'Missing Information',
                    text: 'Please fill in all required fields',
                    confirmButtonColor: '#7aa22e'
                });
                return;
            }
            
            $.ajax({
                url: '@Url.Action("CreateRoom", "Admin")',
                type: 'POST',
                data: data,
                success: function(response) {
                    if (response.success) {
                        Swal.fire({
                            icon: 'success',
                            title: 'Success!',
                            text: response.message,
                            confirmButtonColor: '#7aa22e'
                        }).then(() => location.reload());
                    } else {
                        Swal.fire({
                            icon: 'error',
                            title: 'Error',
                            text: response.message,
                            confirmButtonColor: '#7aa22e'
                        });
                    }
                }
            });
        }

        function toggleRoomStatus(roomId, isActive) {
            const action = isActive ? 'deactivate' : 'activate';
            Swal.fire({
                title: 'Confirm Action',
                text: `Do you want to ${action} this room?`,
                icon: 'warning',
                showCancelButton: true,
                confirmButtonColor: '#7aa22e',
                cancelButtonColor: '#6b7280',
                confirmButtonText: `Yes, ${action}`,
                cancelButtonText: 'Cancel'
            }).then((result) => {
                if (result.isConfirmed) {
                    $.ajax({
                        url: '@Url.Action("ToggleRoomStatus", "Admin")',
                        type: 'POST',
                        data: { roomId: roomId, isActive: !isActive },
                        success: function(response) {
                            if (response.success) {
                                Swal.fire({
                                    icon: 'success',
                                    title: 'Success!',
                                    text: response.message,
                                    confirmButtonColor: '#7aa22e'
                                }).then(() => location.reload());
                            } else {
                                Swal.fire({
                                    icon: 'error',
                                    title: 'Error',
                                    text: response.message,
                                    confirmButtonColor: '#7aa22e'
                                });
                            }
                        }
                    });
                }
            });
        }
