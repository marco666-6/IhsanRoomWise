@section Scripts {
    <script>
        let logsTable;

        function initializeDataTable() {
            logsTable = $('#logsTable').DataTable({
                responsive: false,
                scrollX: true,
                pageLength: 50,
                order: [[5, 'desc']],
                language: {
                    search: "",
                    searchPlaceholder: "üîç Search logs...",
                    lengthMenu: "Show _MENU_ logs",
                    info: "Showing _START_ to _END_ of _TOTAL_ logs",
                    emptyTable: `
                        <div class="text-center py-4 text-muted">
                            <i class="fas fa-history fa-2x mb-2"></i>
                            <p class="mb-0">No activity logs found</p>
                        </div>
                    `,
                    zeroRecords : `
                        <div class="text-center py-4 text-muted">
                            <i class="fas fa-search fa-2x mb-2"></i>
                            <p class="mb-0">No matching logs found</p>
                        </div>
                    `,  
                    paginate: {
                        previous: "‚Äπ",
                        next: "‚Ä∫"
                    }
                },
                dom:
                    "<'row mb-3'<'col-sm-12 col-md-6'l><'col-sm-12 col-md-6'f>>" +
                    "<'row'<'col-sm-12'tr>>" +
                    "<'row mt-3'<'col-sm-12 col-md-5'i><'col-sm-12 col-md-7'p>>",
            });

            // Populate action filter
            const actions = new Set();
            @foreach (var log in Model)
            {
                <text>actions.add('@log.log_action');</text>
            }
            actions.forEach(action => {
                $('#actionFilter').append(`<option value="${action}">${action}</option>`);
            });

            // Populate user filter
            const users = new Set();
            @foreach (var log in Model)
            {
                <text>users.add('@log.user_full_name');</text>
            }
            users.forEach(user => {
                $('#userFilter').append(`<option value="${user}">${user}</option>`);
            });
        }

        $(document).ready(function() {
            initializeDataTable();
            setupFilters();
        });

        function setupFilters() {
            $('#actionFilter, #userFilter, #dateFilter').on('change', function() {
                applyFilters();
            });
        }

        function applyFilters() {
            const action = $('#actionFilter').val();
            const user = $('#userFilter').val();
            const dateRange = $('#dateFilter').val();

            const today = new Date();
            today.setHours(0, 0, 0, 0);

            logsTable.rows().every(function() {
                const row = $(this.node());
                let show = true;

                // Action filter
                if (action && row.data('action') !== action) show = false;

                // User filter
                if (user && row.data('user') !== user) show = false;

                // Date filter
                if (dateRange && show) {
                    const logDate = new Date(row.data('date'));
                    logDate.setHours(0, 0, 0, 0);

                    switch(dateRange) {
                        case 'today':
                            if (logDate.getTime() !== today.getTime()) show = false;
                            break;
                        case 'yesterday':
                            const yesterday = new Date(today);
                            yesterday.setDate(yesterday.getDate() - 1);
                            if (logDate.getTime() !== yesterday.getTime()) show = false;
                            break;
                        case 'week':
                            const weekAgo = new Date(today);
                            weekAgo.setDate(weekAgo.getDate() - 7);
                            if (logDate < weekAgo) show = false;
                            break;
                        case 'month':
                            const monthAgo = new Date(today);
                            monthAgo.setDate(monthAgo.getDate() - 30);
                            if (logDate < monthAgo) show = false;
                            break;
                    }
                }

                row.toggle(show);
            });
        }

        function clearFilters() {
            $('#actionFilter, #userFilter, #dateFilter').val('');
            logsTable.rows().every(function() {
                $(this.node()).show();
            });
        }
    </script>
}

@section Scripts {
    <script>
        function togglePassword(inputId) {
            const input = document.getElementById(inputId);
            const button = input.parentElement.querySelector('.password-toggle');
            const icon = button.querySelector('i');
            
            if (input.type === 'password') {
                input.type = 'text';
                icon.classList.remove('fa-eye');
                icon.classList.add('fa-eye-slash');
            } else {
                input.type = 'password';
                icon.classList.remove('fa-eye-slash');
                icon.classList.add('fa-eye');
            }
        }

        $(document).ready(function() {
            // Profile Form Submission
            $('#profileForm').submit(function(e) {
                e.preventDefault();
                
                const data = {
                    fullName: $('#fullName').val(),
                    email: $('#email').val(),
                    deptName: $('#deptName').val()
                };
                
                $.ajax({
                    url: '@Url.Action("UpdateProfile", ViewContext.RouteData.Values["controller"].ToString())',
                    type: 'POST',
                    data: data,
                    success: function(response) {
                        if (response.success) {
                            Swal.fire({
                                icon: 'success',
                                title: 'Success!',
                                text: response.message,
                                confirmButtonColor: '#7aa22e'
                            }).then(() => {
                                location.reload();
                            });
                        } else {
                            Swal.fire({
                                icon: 'error',
                                title: 'Error',
                                text: response.message,
                                confirmButtonColor: '#7aa22e'
                            });
                        }
                    },
                    error: function() {
                        Swal.fire({
                            icon: 'error',
                            title: 'Error',
                            text: 'An error occurred while updating your profile',
                            confirmButtonColor: '#7aa22e'
                        });
                    }
                });
            });
            
            // Password Form Submission
            $('#passwordForm').submit(function(e) {
                e.preventDefault();
                
                const currentPassword = $('#currentPassword').val();
                const newPassword = $('#newPassword').val();
                const confirmPassword = $('#confirmPassword').val();
                
                if (newPassword !== confirmPassword) {
                    Swal.fire({
                        icon: 'error',
                        title: 'Passwords Don\'t Match',
                        text: 'New password and confirmation password do not match',
                        confirmButtonColor: '#7aa22e'
                    });
                    return;
                }
                
                if (newPassword.length < 6) {
                    Swal.fire({
                        icon: 'error',
                        title: 'Password Too Short',
                        text: 'Password must be at least 6 characters long',
                        confirmButtonColor: '#7aa22e'
                    });
                    return;
                }
                
                $.ajax({
                    url: '@Url.Action("ChangePassword", ViewContext.RouteData.Values["controller"].ToString())',
                    type: 'POST',
                    data: {
                        currentPassword: currentPassword,
                        newPassword: newPassword
                    },
                    success: function(response) {
                        if (response.success) {
                            Swal.fire({
                                icon: 'success',
                                title: 'Password Changed!',
                                text: response.message,
                                confirmButtonColor: '#7aa22e'
                            }).then(() => {
                                $('#passwordForm')[0].reset();
                            });
                        } else {
                            Swal.fire({
                                icon: 'error',
                                title: 'Error',
                                text: response.message,
                                confirmButtonColor: '#7aa22e'
                            });
                        }
                    },
                    error: function() {
                        Swal.fire({
                            icon: 'error',
                            title: 'Error',
                            text: 'An error occurred while changing your password',
                            confirmButtonColor: '#7aa22e'
                        });
                    }
                });
            });
        });
    </script>
}
