<script>
    let roomsTable;
    let lastOpenedModal = null;
    let locations = [];
    let currentView = 'table';

    $(document).ready(function() {
        initializeDataTable();
        loadLocations();
        loadRooms();
        bindEvents();
    });

    function initializeDataTable() {
        roomsTable = $('#roomsTable').DataTable({
            responsive: false,
            scrollX: true,
            pageLength: 25,
            order: [[1, 'asc']],
            language: {
                search: "",
                searchPlaceholder: "üîç Search rooms...",
                lengthMenu: "Show _MENU_ rooms",
                emptyTable: "No rooms found"
            },
            columns: [
                { data: 'room_code', className: "text-center" },
                { data: 'room_name', className: "text-center" },
                { 
                    data: null,
                    className: "text-center",
                    render: function(data) {
                        return `${data.location_plant_name} - B${data.location_block} - F${data.location_floor}`;
                    }
                },
                { 
                    data: 'room_capacity',
                    className: "text-center",
                    render: function(data) {
                        return `<i class="fas fa-users"></i> ${data}`;
                    }
                },
                { 
                    data: 'facilities',
                    className: "text-center",
                    render: function(data) {
                        let facilities = [];
                        if (data.projector) facilities.push('<i class="fas fa-video" title="Projector"></i>');
                        if (data.smart_screen) facilities.push('<i class="fas fa-tv" title="Smart Screen"></i>');
                        if (data.screenbeam) facilities.push('<i class="fas fa-wifi" title="ScreenBeam"></i>');
                        if (data.cisco_bar) facilities.push('<i class="fas fa-phone" title="CiscoBar"></i>');
                        return facilities.join(' ') || '-';
                    }
                },
                { 
                    data: 'room_operational_status',
                    className: "text-center",
                    render: function(data) {
                        const statusMap = {
                            'Available': 'bg-success',
                            'Maintenance': 'bg-warning',
                            'Unavailable': 'bg-danger'
                        };
                        return `<span class="badge ${statusMap[data]}">${data}</span>`;
                    }
                },
                { 
                    data: null,
                    orderable: false,
                    className: "text-center",
                    render: function(data, type, row) {
                        return `
                            <div style="display:flex; gap:6px; justify-content:center;">
                                <button class="btn btn-sm btn-outline-primary" onclick="editRoom(${row.room_id})" title="Edit">
                                    <i class="fas fa-edit"></i>
                                </button>
                                <button class="btn btn-sm btn-outline-success" onclick="changeStatus(${row.room_id}, '${row.room_operational_status}')" title="Change Status">
                                    <i class="fas fa-exchange-alt"></i>
                                </button>
                            </div>
                        `;
                    }
                }
            ]
        });
    }

    function bindEvents() {
        $('#searchInput').on('keyup', function() {
            loadRooms();
        });

        $('#statusFilter, #locationFilter, #capacityFilter').on('change', function() {
            loadRooms();
        });

        $('#tableViewBtn').click(function() {
            currentView = 'table';
            $('#tableView').show();
            $('#cardsView').hide();
            $(this).addClass('active');
            $('#cardsViewBtn').removeClass('active');
        });

        $('#cardsViewBtn').click(function() {
            currentView = 'cards';
            $('#tableView').hide();
            $('#cardsView').show();
            $(this).addClass('active');
            $('#tableViewBtn').removeClass('active');
            renderCards();
        });

        $('#addRoomBtn').click(function() {
            $('#roomForm')[0].reset();
            $('#roomId').val('');
            $('#modalTitle').text('Add New Room');
            $('#roomModal').modal('show');
        });

        $('#saveRoomBtn').click(saveRoom);
        $('#exportBtn').click(exportRooms);

        $('#manageLocationBtn').click(function() {
            lastOpenedModal = '#roomModal';
            $('#roomModal').modal('hide');
            $('#roomModal').one('hidden.bs.modal', function() {
                loadLocationsTable();
                $('#locationModal').modal('show');
            });
        });

        $('#addLocationBtn').click(function() {
            lastOpenedModal = '#locationModal';
            $('#locationModal').modal('hide');
            $('#locationModal').one('hidden.bs.modal', function() {
                $('#locationForm')[0].reset();
                $('#locationId').val('');
                $('#locationModalTitle').text('Add Location');
                $('#locationFormModal').modal('show');
            });
        });

        $('#saveLocationBtn').click(saveLocation);
    }

    function loadRooms() {
        const search = $('#searchInput').val();
        const status = $('#statusFilter').val();
        const locationId = $('#locationFilter').val();
        const capacity = $('#capacityFilter').val();
        
        let minCapacity = null, maxCapacity = null;
        if (capacity) {
            const range = capacity.split('-');
            minCapacity = parseInt(range[0]);
            maxCapacity = parseInt(range[1]);
        }

        $.ajax({
            url: '/Admin/GetRooms',
            method: 'GET',
            data: { search, status, locationId, minCapacity, maxCapacity },
            success: function(response) {
                if (response.success) {
                    roomsTable.clear();
                    roomsTable.rows.add(response.data);
                    roomsTable.draw();
                    updateStats(response.data);
                    if (currentView === 'cards') {
                        renderCards();
                    }
                }
            }
        });
    }

    function updateStats(rooms) {
        $('#totalRooms').text(rooms.length);
        $('#availableRooms').text(rooms.filter(r => r.room_operational_status === 'Available').length);
        $('#maintenanceRooms').text(rooms.filter(r => r.room_operational_status === 'Maintenance').length);
    }

    function renderCards() {
        const container = $('#cardsContainer');
        container.empty();
        
        roomsTable.rows().every(function() {
            const room = this.data();
            const statusClass = {
                'Available': 'status-available',
                'Maintenance': 'status-maintenance',
                'Unavailable': 'status-unavailable'
            }[room.room_operational_status];

            const facilities = [];
            if (room.facilities.projector) facilities.push('<span class="facility-badge"><i class="fas fa-video"></i> Projector</span>');
            if (room.facilities.smart_screen) facilities.push('<span class="facility-badge"><i class="fas fa-tv"></i> Smart Screen</span>');
            if (room.facilities.screenbeam) facilities.push('<span class="facility-badge"><i class="fas fa-wifi"></i> ScreenBeam</span>');
            if (room.facilities.cisco_bar) facilities.push('<span class="facility-badge"><i class="fas fa-phone"></i> CiscoBar</span>');

            container.append(`
                <div class="room-card">
                    <div class="room-card-image">
                        <i class="fas fa-door-open"></i>
                    </div>
                    <div class="room-card-body">
                        <div class="room-card-title">${room.room_name}</div>
                        <div class="room-card-code">${room.room_code}</div>
                        <div class="room-info-item">
                            <i class="fas fa-map-marker-alt"></i>
                            ${room.location_plant_name} - Block ${room.location_block} - Floor ${room.location_floor}
                        </div>
                        <div class="room-info-item">
                            <i class="fas fa-users"></i>
                            Capacity: ${room.room_capacity} people
                        </div>
                        <div class="room-facilities">
                            ${facilities.join('')}
                        </div>
                        <div style="margin-top: 1rem;">
                            <span class="status-badge ${statusClass}">
                                <i class="fas fa-circle" style="font-size: 0.5rem;"></i>
                                ${room.room_operational_status}
                            </span>
                        </div>
                        <div class="action-buttons" style="margin-top: 1rem;">
                            <button class="btn btn-sm btn-primary" onclick="editRoom(${room.room_id})" style="flex: 1;">
                                <i class="fas fa-edit"></i> Edit
                            </button>
                            <button class="btn btn-sm btn-secondary" onclick="changeStatus(${room.room_id}, '${room.room_operational_status}')" style="flex: 1;">
                                <i class="fas fa-exchange-alt"></i> Status
                            </button>
                        </div>
                    </div>
                </div>
            `);
        });
    }

    function loadLocations() {
        $.ajax({
            url: '/Admin/GetLocations',
            method: 'GET',
            success: function(response) {
                if (response.success) {
                    locations = response.data;
                    const locationSelect = $('#roomLocation, #locationFilter');
                    locationSelect.find('option:not(:first)').remove();
                    response.data.forEach(loc => {
                        $('#roomLocation').append(`<option value="${loc.location_id}">${loc.display_name}</option>`);
                        $('#locationFilter').append(`<option value="${loc.location_id}">${loc.display_name}</option>`);
                    });
                }
            }
        });
    }

    function saveRoom() {
        if (!$('#roomForm')[0].checkValidity()) {
            $('#roomForm')[0].reportValidity();
            return;
        }

        const roomId = $('#roomId').val();
        const roomData = {
            room_id: roomId ? parseInt(roomId) : 0,
            room_name: $('#roomName').val(),
            room_location_id: parseInt($('#roomLocation').val()),
            room_capacity: parseInt($('#roomCapacity').val()),
            room_description: $('#roomDescription').val() || null,
            room_has_projector: $('#hasProjector').is(':checked'),
            room_has_smart_screen: $('#hasSmartScreen').is(':checked'),
            room_has_screenbeam: $('#hasScreenbeam').is(':checked'),
            room_has_cisco_bar: $('#hasCiscoBar').is(':checked'),
            room_other_facilities: $('#otherFacilities').val() || null,
            room_operational_status: $('#operationalStatus').val()
        };

        $.ajax({
            url: '/Admin/CreateRoom',
            method: 'POST',
            contentType: 'application/json',
            data: JSON.stringify(roomData),
            success: function(response) {
                if (response.success) {
                    Swal.fire('Success', response.message, 'success');
                    $('#roomModal').modal('hide');
                    loadRooms();
                } else {
                    Swal.fire('Error', response.message, 'error');
                }
            }
        });
    }

    function editRoom(roomId) {
        roomsTable.rows().every(function() {
            const data = this.data();
            if (data.room_id === roomId) {
                $('#roomId').val(data.room_id);
                $('#roomName').val(data.room_name);
                $('#roomLocation').val(data.room_location_id);
                $('#roomCapacity').val(data.room_capacity);
                $('#roomDescription').val(data.room_description || '');
                $('#hasProjector').prop('checked', data.room_has_projector);
                $('#hasSmartScreen').prop('checked', data.room_has_smart_screen);
                $('#hasScreenbeam').prop('checked', data.room_has_screenbeam);
                $('#hasCiscoBar').prop('checked', data.room_has_cisco_bar);
                $('#otherFacilities').val(data.room_other_facilities || '');
                $('#operationalStatus').val(data.room_operational_status);
                $('#modalTitle').text('Edit Room');
                $('#roomModal').modal('show');
                return false;
            }
        });
    }

    function changeStatus(roomId, currentStatus) {
        Swal.fire({
            title: 'Change Room Status',
            input: 'select',
            inputOptions: {
                'Available': 'Available',
                'Maintenance': 'Maintenance',
                'Unavailable': 'Unavailable'
            },
            inputValue: currentStatus,
            showCancelButton: true,
            confirmButtonColor: '#7aa22e'
        }).then((result) => {
            if (result.isConfirmed) {
                $.ajax({
                    url: '/Admin/UpdateRoomStatus',
                    method: 'POST',
                    data: { roomId: roomId, status: result.value },
                    success: function(response) {
                        if (response.success) {
                            Swal.fire('Success', response.message, 'success');
                            loadRooms();
                        }
                    }
                });
            }
        });
    }

    function exportRooms() {
        window.location.href = '/Admin/ExportRooms';
    }

    function loadLocationsTable() {
        $.ajax({
            url: '/Admin/GetLocations',
            method: 'GET',
            success: function(response) {
                if (response.success) {
                    const tbody = $('#locationsTableBody');
                    tbody.empty();
                                        response.data.forEach(loc => {
                        tbody.append(`
                            <tr>
                                <td>${loc.location_code}</td>
                                <td>${loc.plant_name}</td>
                                <td>${loc.block}</td>
                                <td>${loc.floor}</td>
                                <td>
                                    <div style="display:flex; gap:6px; justify-content:center;">
                                        <button class="btn btn-sm btn-outline-primary"
                                                onclick="editLocation(${loc.location_id})">
                                            <i class="fas fa-edit"></i>
                                        </button>
                                        <button class="btn btn-sm btn-outline-danger"
                                                onclick="deleteLocation(${loc.location_id})">
                                            <i class="fas fa-trash"></i>
                                        </button>
                                    </div>
                                </td>
                            </tr>
                        `);
                    });
                }
            }
        });
    }

    function saveLocation() {
        if (!$('#locationForm')[0].checkValidity()) {
            $('#locationForm')[0].reportValidity();
            return;
        }

        const locationData = {
            location_id: $('#locationId').val() ? parseInt($('#locationId').val()) : 0,
            plant_name: $('#plantName').val(),
            block: parseInt($('#block').val()),
            floor: parseInt($('#floor').val()),
            description: $('#locationDescription').val() || null
        };

        $.ajax({
            url: '/Admin/CreateLocation',
            method: 'POST',
            contentType: 'application/json',
            data: JSON.stringify(locationData),
            success: function(response) {
                if (response.success) {
                    Swal.fire('Success', response.message, 'success');
                    $('#locationFormModal').modal('hide');

                    loadLocations();

                    if (lastOpenedModal) {
                        $(lastOpenedModal).modal('show');
                        lastOpenedModal = null;
                    }
                } else {
                    Swal.fire('Error', response.message, 'error');
                }
            }
        });
    }

    function editLocation(locationId) {
        $.ajax({
            url: '/Admin/GetLocationById',
            method: 'GET',
            data: { id: locationId },
            success: function(response) {
                if (response.success) {
                    const loc = response.data;

                    $('#locationId').val(loc.location_id);
                    $('#plantName').val(loc.plant_name);
                    $('#block').val(loc.block);
                    $('#floor').val(loc.floor);
                    $('#locationDescription').val(loc.description || '');
                    $('#locationModalTitle').text('Edit Location');

                    lastOpenedModal = '#locationModal';
                    $('#locationModal').modal('hide');

                    $('#locationModal').one('hidden.bs.modal', function () {
                        $('#locationFormModal').modal('show');
                    });
                }
            }
        });
    }

    function deleteLocation(locationId) {
        Swal.fire({
            title: 'Delete Location?',
            text: 'This action cannot be undone',
            icon: 'warning',
            showCancelButton: true,
            confirmButtonColor: '#dc2626'
        }).then((result) => {
            if (result.isConfirmed) {
                $.ajax({
                    url: '/Admin/DeleteLocation',
                    method: 'POST',
                    data: { id: locationId },
                    success: function(response) {
                        if (response.success) {
                            Swal.fire('Deleted', response.message, 'success');
                            loadLocationsTable();
                            loadLocations();
                        }
                    }
                });
            }
        });
    }
</script>
}
